 14 vui.isArray = function(obj){

 16 }

 18 /**
 19  * @name vui.isEmpty
 20  * @param {All} obj 主体
 21  * @return {Boolean} true/false
 22  */
 23 vui.isEmpty = function (obj) {
 24     return obj === null || typeof obj === 'undefined' || 
 25 			 obj === 0 || obj === false || obj === '' || 
 26 			(typeof obj.length === 'number' && obj.length === 0);
 27   };
